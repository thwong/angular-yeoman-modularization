'use strict';
 
/**
  * @ngdoc overview
  * @name angularYeomanModularizationApp
  * @description
  * # angularYeomanModularizationApp
  *
  * Main module of the application.
 */
angular.module('angularYeomanModularizationApp', ['ui.router', 'oc.lazyLoad']).config(function($locationProvider) {
  return $locationProvider.html5Mode(true).hashPrefix('!');
}).config(function($stateProvider, $urlRouterProvider) {
  $urlRouterProvider.otherwise("/");
  return $stateProvider.state('mainState', {
    url: '/',
    views: {
      '': {
        controller: 'MainCtrl',
        templateUrl: 'views/main.html'
      }
    },
    resolve: {
      files: function($ocLazyLoad) {
        return $ocLazyLoad.load('scripts/controllers/main.js');
      }
    }
  });
});
